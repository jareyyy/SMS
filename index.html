// api/smsProxy.js export default async function handler(req, res) { const apiUrl = 'https://api.kenliejugarap.com/freesmslbc/'; if (req.method === 'OPTIONS') { // Handle preflight requests res.setHeader('Access-Control-Allow-Origin', '*'); // Change '*' to your specific domain for security res.setHeader('Access-Control-Allow-Methods', 'POST, GET, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type'); res.status(200).end(); return; } if (req.method === 'POST') { const { number, message } = req.body; try { const response = await fetch(apiUrl, { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ number, message }), }); const data = await response.json(); res.setHeader('Access-Control-Allow-Origin', '*'); // Change '*' to your specific domain for security res.status(response.status).json(data); } catch (error) { console.error('Error sending SMS:', error); res.setHeader('Access-Control-Allow-Origin', '*'); // Change '*' to your specific domain for security res.status(500).json({ status: false, response: 'Failed to send SMS.' }); } } else { res.setHeader('Access-Control-Allow-Origin', '*'); // Change '*' to your specific domain for security res.status(405).json({ error: 'Method not allowed' }); } }
